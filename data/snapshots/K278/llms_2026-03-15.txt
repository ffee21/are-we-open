function moveToSearchPage() {
	var query = jQuery('#keyWord');
	var keyword = query.val();
	
	if(keyword ==""  || keyword =="undefined" || keyword == null){
   	  alert("검색어를 입력해주세요.");
   	  return false;
     }

	location.href='/iwt/ds/ko/information/totalSearch.do?searchKeyword=' + keyword;
			
}

function keyevent() {
	var keycode = event.keyCode;
	if(keycode == 13) {
		moveToSearchPage();
	}
}

$(document).ready(function() {
	var dupYn = jQuery("#dupYn").val();
	var currentSessionId = jQuery("#currentSessionId").val();
	
	
	if(dupYn.length == 0 && currentSessionId.length == 0){
		return true;
	} else if(currentSessionId.length > 0) {
		
		$.ajax({
			type:'POST',
			url:'/com/cmm/web/duplicateLoginCheck.ajax',
			data:{sessionId : currentSessionId},
			contentType: "application/x-www-form-urlencoded; charset=utf-8",
			dataType:'json',
			success:function(result) {
				console.log(result);
				if(result.result == 'N') {
					location.href = "/com/cmm/web/duplicatelogin.do";
				} else {
					return true;
				}
			},
			error:function(xhr, textStatus, result) {
				alert(xhr.status + " : " + xhr.statusText + " : " + textStatus);
			}
		}); 
	}
});
