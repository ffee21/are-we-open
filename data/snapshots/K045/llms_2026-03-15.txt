let containerLast = false;
// 접근성 focus 항목 탐색
function focusAccessibility() {
	document.querySelector('#container').removeEventListener("keydown", keyDownTab);
	containerLast = false;
	document.querySelector('#container').addEventListener("keydown", keyDownTab);
}
// 탭 클릭 이벤트(접근성)
function keyDownTab(e) {
	const focusItems = $('#container a:visible, #container button:visible, #container *[tabindex=0]');
	if (e.key === "Tab" && !e.shiftKey) {
		if(containerLast) {
			e.preventDefault();
			$('#sideMenu a').first().focus();
			containerLast = false;
		}
		if(document.activeElement == focusItems[focusItems.length -2]) {
			containerLast = true;
		}
	}
}

function openReserve( deptCd, profEmpCd ) {
	
	if( !deptCd ) deptCd = "";
	if( !profEmpCd ) profEmpCd = "";
	
	var _width = '1920';
	var _height = '1100';

	var xPos = (document.body.offsetWidth/2)-(_width/2);
	xPos += window.screenLeft;	// 듀얼모니터일 때
	var yPos = (document.body.offsetHeight/2) - (_height/2) + 50;
	
	var isMobile = /iPhone|iPad|iPod|Android/i.test(navigator.userAgent);
	
	if( isMobile == true ) {
	
		location.href = "/nmc/online/reservePop?deptCd=" + deptCd + "&profEmpCd=" + profEmpCd;
		
	} else {
	
		// 인터넷 진료예약
		var pop = window.open("/nmc/online/reservePop?deptCd=" + deptCd + "&profEmpCd=" + profEmpCd, 'openReserve', "width=" + _width + ", height=" + _height + ", left=" + xPos + ", top=" + yPos + ", menubar=no, toolbar=no, location=no, directories=no, status=no, scrollbars=yes, copyhistory=no, resizable=yes");
		
		if( pop == null ){
			alert("팝업이 차단 되었습니다.\n차단을 해제해주세요.");
		}
	
	}
		
}

function goMemReserve( deptCd, profEmpCd ){
	
	var patNo = "";
	if( patNo ) {
		
		$.ajax({
			type: "POST",
			url: "/nmc/online/verifyProc",
			data: { 
				"patNo": patNo,
				"dutyYn": "N",
				"memberYn": "Y",
				"zipCode": "",
				"buildNo": "",
				"addr1": "",
				"addr2": "",			
				"mobile1": "",
				"mobile2": "",
				"mobile3": "",
				"ssn1": "",
				"ssn2": ""
			},
			success: function( res ){
				
				if( res ){
					
					alert( res );
					return;
						
				}
				
				// 진료예약 팝업
				openReserve(deptCd, profEmpCd);
				
			}
		});
		
	} else {
		
		location.href = "/nmc/online/verifyMe?deptCd=" + deptCd + "&profEmpCd=" + profEmpCd;
		
	}
	
}

document.addEventListener("DOMContentLoaded", () => {
	const menuWraps = document.querySelectorAll(".gnbMenuWrap");
	const menuListWrap = document.querySelector('.gnbWrap ul');

	menuWraps.forEach((wrap, index) => {
		const secondDepth = wrap.querySelector(".second_depth");
		if (!secondDepth) return;
		const links = Array.from(secondDepth.querySelectorAll("a"));

		const showMenu = () => {
			secondDepth.style.display = "block";
			secondDepth.style.maxHeight = secondDepth.scrollHeight + "px";
		};

		const hideMenu = () => {
			secondDepth.style.maxHeight = "0";
			secondDepth.style.display = "none";
		};

		// 마우스 이벤트
		wrap.addEventListener("mouseover", showMenu);
		menuListWrap.addEventListener("mouseout", hideMenu);

		// 키보드 focus 이벤트
		wrap.addEventListener("focusin", showMenu);
		wrap.addEventListener("focusout", (e) => {
			if (!wrap.contains(e.relatedTarget)) {
				hideMenu();
			}
		});

		links.forEach((link) => {
			link.addEventListener("focusin", showMenu);
			link.addEventListener("focusout", (e) => {
				if (!wrap.contains(e.relatedTarget)) {
					hideMenu();
				}
			});
		});
	});
});
