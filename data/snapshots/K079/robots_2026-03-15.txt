		anyidAdaptor.userCheck = function () {
			//console.log(">>> userCheck 함수 진입");
			var xhr = new XMLHttpRequest();
			//console.log(">>> xhr 객체 생성 완료");
			xhr.open("POST", "/oidc/userCheck", true);
			//console.log(">>> xhr.open 호출 완료");
			xhr.setRequestHeader("Content-Type", "application/x-www-form-urlencoded");
			xhr.onreadystatechange = function () {
				if (xhr.readyState !== 4) return;

				//console.log("status:", xhr.status);
				//console.log("responseText(raw):", xhr.responseText);
				if (xhr.status === 200) {
					//바디가 없으면 바로 리턴
					if (!xhr.responseText || xhr.responseText.trim() === "") {
						//console.warn("빈 응답입니다.");
						return;
					}
					try {
							var response = JSON.parse(xhr.responseText);
							if (response && response.success) {
								//SSO 로그인 성공
								anyidAdaptor.ssoLogin();
							} else {
								setTimeout(function () {
								//andId 가입안되어 있으면 가입 팝업창 띄움
								var uri  = (response && response.uri)  ? response.uri  : undefined;
								var data = (response && response.data) ? response.data : undefined;
								anyidAdaptor.receiveAnyIdJoin(uri, data);
							}, 100);
						}
					} catch (e) {
							//console.log("error : " + e);
							alert("오류가 발생하였습니다.\n관리자에게 문의하세요.!");	
					}
				} else {
					alert("오류가 발생하였습니다.\n관리자에게 문의하세요..");	
				}
			};
			const jsonObject = {
				txId: anyidAdaptor.certData.txId,
				ssob: anyidAdaptor.certData.ssob,
				userSeCd: anyidAdaptor.certData.userSeCd,
			};
			const encodedString = btoa(JSON.stringify(jsonObject));
			const data = "data=" + encodeURIComponent(encodedString) + "&txId=" + encodeURIComponent(anyidAdaptor.certData.txId);
			xhr.send(data);
		};

        anyidAdaptor.receiveAnyIdJoin = function(portalJoinUri, memberData){

            var url = portalJoinUri+encodeURIComponent(memberData);
            anyidAdaptor.popup(url,"openAnyIdRegistUserIdv",700,740);

            window.addEventListener("message", function (event) {

			var result;

			try {
				if (typeof event.data === "string") {
					result = JSON.parse(event.data);     // JSON 문자열이면 파싱
				} else if (typeof event.data === "object" && event.data !== null) {
					result = event.data;                 // 이미 객체면 그대로 사용
				} else {
					//console.log("Unexpected message data:", event.data);
					return;
				}
			} catch (e) {
				//console.log("message parse error:", e, "raw:", event.data);
				alert("오류가 발생하였습니다.\n관리자에게 문의하세요.");
				return;
			}

			if (!result.funcName || (result.funcName !== "trmsAgreComplete" && result.funcName !== "registUserComplete")) {
				return;
			}

			if (result.status === "success") {
				if (result.trmsAgreYn == "Y") {
					anyidAdaptor.ssoLogin();
				} else {
					anyidAdaptor.orgLogin(anyidAdaptor.certData);
				}
			} else {
				anyidAdaptor.orgLogin(anyidAdaptor.certData);
			}

		}, false);
        }

                popup.focus();
            }
        }

        anyidAdaptor.ssoLogin = function(){

            jsonObject.ssob = anyidAdaptor.certData.ssob;
            jsonObject.userSeCd = anyidAdaptor.certData.userSeCd;

        anyidAdaptor.ssoLoginPageSub = function(baseUrl, endPoint, acrValues){
            var subUrl = "";

            if (endPoint && endPoint.trim() !== "") {
                subUrl = "?endPoint=" + encodeURIComponent(endPoint);
            }

            if (!acrValues && acrValues.trim() == "") {
                acrValues = "3";

            if (acrValues && acrValues.trim() !== "") {
                if(subUrl == ""){
                    subUrl = "?";
                }else{
                    subUrl = subUrl + "&";
                }
                subUrl = subUrl + "acrValues=" + acrValues;
            }

            window.location.replace(baseUrl+subUrl);
        }
