  // 전체메뉴 접근성 개선: aria-expanded / aria-hidden 동기화, 포커스 관리, Esc 닫기
  (function() {
    const allMenuWrap = document.getElementById('all-menu-wrap');
    const toggleBtn = allMenuWrap ? allMenuWrap.querySelector('button.header-btn:not(.all-menu-close)') : null;
    const closeBtn = allMenuWrap ? allMenuWrap.querySelector('button.all-menu-close') : null;
    const panel = document.getElementById('all-menu-subnav');

      function getFocusable() {
          return Array.from(
              allMenuWrap.querySelectorAll(
                  'button.all-menu-close, a[href], button:not([disabled]), input, [tabindex]:not([tabindex="-1"])'
              )
          ).filter(el => {
              return el.offsetParent !== null;
          });
      }

      let isTrapBound = false;

      function trapFocus() {
          if (isTrapBound) return;
          isTrapBound = true;

          document.addEventListener('keydown', handleTrap);
      }

      function releaseTrap() {
          document.removeEventListener('keydown', handleTrap);
          isTrapBound = false;
      }

      function handleTrap(e) {
          if (e.key !== 'Tab') return;
          if (panel.getAttribute('aria-hidden') === 'true') return;

          const focusable = getFocusable();
          if (!focusable.length) return;

          const first = focusable[0];
          const last  = focusable[focusable.length - 1];

          if (e.shiftKey) {
              if (document.activeElement === first) {
                  e.preventDefault();
                  last.focus();
              }
          } else {
              if (document.activeElement === last) {
                  e.preventDefault();
                  first.focus();
              }
          }
      }


      function openPanel() {
      if (!allMenuWrap || !toggleBtn || !panel || !closeBtn) return;
      allMenuWrap.classList.add('active');
      document.body.classList.add('no-scroll');
      toggleBtn.setAttribute('aria-expanded', 'true');
      panel.setAttribute('aria-hidden', 'false');

      // 토글 버튼 숨기고 닫기 버튼 표시
      toggleBtn.style.display = 'none';
      closeBtn.style.display = '';

      // 포커스 가능한 첫 요소로 포커스 이동
      // const first = panel.querySelector('a, button, input, [tabindex]:not([tabindex="-1"])');
      // if (first) first.focus();
      // else {
      //   panel.setAttribute('tabindex', '-1');
      //   panel.focus();
      // }

      closeBtn.focus();

      trapFocus(allMenuWrap);
    }

    function closePanel() {
      if (!allMenuWrap || !toggleBtn || !panel || !closeBtn) return;
      allMenuWrap.classList.remove('active');
      document.body.classList.remove('no-scroll');
      toggleBtn.setAttribute('aria-expanded', 'false');
      panel.setAttribute('aria-hidden', 'true');
      if (panel.hasAttribute('tabindex')) panel.removeAttribute('tabindex');

      // 닫기 버튼 숨기고 토글 버튼 표시
      closeBtn.style.display = 'none';
      toggleBtn.style.display = '';

      // 포커스 복귀
      releaseTrap();
      toggleBtn.focus();
    }

    if (toggleBtn) {
      toggleBtn.addEventListener('click', () => {
        openPanel();
      });
    }

    if (closeBtn) {
      closeBtn.addEventListener('click', () => {
        closePanel();
      });
    }

    // Esc 키로 닫기
    document.addEventListener('keydown', (e) => {
      if (e.key === 'Escape' || e.key === 'Esc') {
        if (panel && panel.getAttribute('aria-hidden') === 'false') closePanel();
      }
    });
  })();

  // 현재 URL 기준으로 메뉴 활성화
  function activateGnbMenu() {
    let currentPath = window.location.pathname;

    // Context path 제거 (JSP에서 주입)
    const contextPath = '';

    if (contextPath && currentPath.startsWith(contextPath)) {
      currentPath = currentPath.substring(contextPath.length);
    }

    // 경로가 비어있거나 홈이면 활성화 안 함
    if (!currentPath || currentPath === '/') {
      return;
    }

    // 현재 URL에서 대분류 추출 (첫 번째 세그먼트)
    // 예: /major-business/donation/organ → major-business
    const pathSegments = currentPath.split('/').filter(s => s);
    const currentCategory = pathSegments[0]; // 대분류

    if (!currentCategory) {
      return;
    }

    // 모든 header-menu-item을 순회하며 하위 메뉴의 경로를 확인
    const menuItems = document.querySelectorAll('.header-menu > .header-menu-item');

    let matched = false;

    menuItems.forEach((menuItem) => {
      // 해당 대분류의 하위 메뉴들의 href를 확인
      const subNavLinks = menuItem.querySelectorAll('.sub-nav a');
      
      subNavLinks.forEach(subLink => {
        if (matched) return;
        
        let subHref = subLink.getAttribute('href') || '';

        // Context path 제거
        if (contextPath && subHref.startsWith(contextPath)) {
          subHref = subHref.substring(contextPath.length);
        }

        // 하위 메뉴의 대분류 추출
        const subSegments = subHref.split('/').filter(s => s);
        const subCategory = subSegments[0];

        // 현재 URL의 대분류와 하위 메뉴의 대분류가 일치하면 활성화
        if (subCategory && subCategory === currentCategory) {
          menuItem.classList.add('active');
          matched = true;
        }
      });
    });
  }

  // 페이지 로드 시 활성화
  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', activateGnbMenu);
  } else {
    activateGnbMenu();
  }
</script>

            function initRelatedSite() {
                const footer = document.getElementById('krds-footer');
                if (!footer) return;

                const linkBtn = footer.querySelector('.foot-quick .link');
                const relatedSite = footer.querySelector('.related-site');
                const closeBtn = footer.querySelector('.related-site .btn-close');

                if (linkBtn && relatedSite) {
                    // 이미 이벤트가 연결되어 있는지 확인
                    if (linkBtn.dataset.listenerAttached === 'true') {
                        return;
                    }

                    // 열기 버튼 클릭 이벤트
                    // linkBtn.addEventListener('click', function(e) {
                    //     e.preventDefault();
                    //     relatedSite.classList.add('show');
                    // });

                    // 열기/닫기 토글 기능 적용
                    linkBtn.addEventListener('click', function(e) {
                        e.preventDefault();
                        relatedSite.classList.toggle('show');   // 토글로 변경
                    });

                    // 닫기 버튼 클릭 이벤트
                    if (closeBtn) {
                        closeBtn.addEventListener('click', function(e) {
                            e.preventDefault();
                            relatedSite.classList.remove('show');
                        });
                    }
                    document.addEventListener('keydown', function (e){
                        if(e.key === 'Escape'){
                            relatedSite.classList.remove('show');
                        }
                    })



                    // 이벤트가 연결되었음을 표시
                    linkBtn.dataset.listenerAttached = 'true';
                }
            }

            function initTabs() {
                const footer = document.getElementById('krds-footer');
                if (!footer) return;

                const tabBtn = footer.querySelectorAll('.tab-btn');
                const tabContent = footer.querySelectorAll('.tab-content-item');

                if (tabBtn.length === 0 || tabContent.length === 0) {
                    return;
                }

                tabBtn.forEach((btn, index) => {
                    // 이미 이벤트가 연결되어 있는지 확인
                    if (btn.dataset.listenerAttached === 'true') {
                        return;
                    }

                    btn.addEventListener('click', function() {
                        // 모든 탭 버튼에서 active 클래스 및 aria-selected 제거
                        tabBtn.forEach(b => {
                            b.classList.remove('active');
                            b.setAttribute('aria-selected', 'false');
                        });
                        // 모든 탭 콘텐츠에서 active 클래스 제거
                        tabContent.forEach(content => {
                            content.classList.remove('active');
                        });
                        // 클릭된 버튼에 active 클래스 및 aria-selected 추가
                        this.classList.add('active');
                        this.setAttribute('aria-selected', 'true');
                        // 해당 인덱스의 탭 콘텐츠에 active 클래스 추가
                        if (tabContent[index]) {
                            tabContent[index].classList.add('active');
                        }
                    });

                    // 이벤트가 연결되었음을 표시
                    btn.dataset.listenerAttached = 'true';
                });
            }

            function initFooter() {
                initRelatedSite();
                initTabs();
            }

            // footer가 이미 로드되어 있으면 즉시 실행
            if (document.getElementById('krds-footer')) {
                initFooter();
            }

            // footer가 나중에 로드될 경우를 대비해 MutationObserver 사용
            const observer = new MutationObserver(function(mutations) {
                const footer = document.getElementById('krds-footer');
                if (footer && footer.querySelectorAll('.tab-btn').length > 0) {
                    initFooter();
                    observer.disconnect();
                }
            });

            observer.observe(document.body, {
                childList: true,
                subtree: true
            });
        })();

    // 쿠키 조회
    function getCookie(name) {
        const cookies = document.cookie ? document.cookie.split(';') : [];
        for (let i = 0; i < cookies.length; i++) {
            const c = cookies[i].trim();
            if (c.indexOf(name + '=') === 0) {
                return c.substring((name + '=').length, c.length);
            }
        }
        return null;
    }

    // 쿠키 설정 (하루: max-age 사용)
    function setCookie(name, value) {
